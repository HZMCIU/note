### AC代码

```cpp
class Solution {
public:
//left>0 || used<=n
    vector<string> generateParenthesis(int n) {
        this->dep=n;
        dfs(0);
        return this->ans;
    }
    
    void dfs(int cur){
        if(temp.length()==2*this->dep){
            ans.push_back(temp);
            return ;
        }
        if(cur<this->dep){
            temp+="(";
            left+=1;
            dfs(cur+1);
            left-=1;
            temp.pop_back();
        }
        if(left>0){
            temp+=")";
            left-=1;
            dfs(cur);
            left+=1;
            temp.pop_back();
        }
    }
private:
    vector<string> ans;
    int left=0;
    string temp="";  
    int dep=0;
};
```

```python
class Solution:
    def generateParenthesis(self, n: int) -> List[str]:
        ans = []
        def f(l, r, s):
            l == r == n and ans.append(s)
            l < n and f(l + 1, r, s + '(')
            r < l and f(l, r + 1, s + ')')
        f(0, 0, '')
        return ans
```
