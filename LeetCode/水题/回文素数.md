#### [回文素数](https://leetcode-cn.com/problems/prime-palindrome/)

### 思路

​	枚举所有的回文数，检测是否为素数。

### AC代码

```cpp
class Solution
{
public:
    int primePalindrome(int N)
    {
        for (long i = 1; i < 10; i++)
        {
            if (isPrime(i) && i >= N)
                return i;
        }
        for (long i = 1; i <= 100000; i++)
        {
            long t = reverse(i, i / 10);
            if (isPrime(t) && t >= N)
                return t;
            t = reverse(i, i);
            if (isPrime(t) && t >= N)
                return t;
        }
        return -1;
    }
    bool isPrime(long n)
    {
        for (long i = 2; i * i <= n; i++)
        {
            if (n % i == 0)
                return false;
        }
        return n != 1;
    }
    long reverse(long x, long y)
    {
        while (y > 0)
        {
            x = x * 10 + (y % 10);
            y /= 10;
        }
        return x;
    }
};
```
